def solve(): N = int(input()) a = list(map(int, input().split())) b = list(map(int, input().split())) result = [0] * (N + 1) for l in range(N): for r in range(l, N): checked_cows = 0 for i in range(N): if l <= i <= r: # 반전된 부분 if a[l + r - i] == b[i]: checked_cows += 1 else: if a[i] == b[i]: checked_cows += result[checked_cows] += 1 for count in result: print(count)
